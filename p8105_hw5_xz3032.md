``` r
ttest = function(mu) {
  sample = tibble(x = rnorm(30, mean = mu, sd = 5))
  sample %>%
    t.test(mu = 0, alternative = "two.sided", conf.level = 0.95) %>%
    broom::tidy() %>%
    select(estimate, p.value)
}

ttest(0)
```

sim_diff_mu = 
  tibble(mu = c(1, 2, 3, 4, 5, 6)) %>%

    t_output = map(.x = mu, ~rerun(5000,ttest(.x))),
